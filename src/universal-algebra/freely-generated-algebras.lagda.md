## Idea

For any [algebraic theory](universal-algebra.algebraic-theories.md) `T` and
generator type `G`, there is an [algebra](universal-algebra.algebras.md) `FG`
satisfying the theory freely generated by `G`, satisfying the universal property
that for any algebra `H` for the theory `T` and any map `f : G → H`, there is a
unique map `φ : FG → H` such that `f = φ ∘ in-generator-free-Algebra`.

## Definition

```agda
module _
  {l1 l2 l3 : Level}
  (σ : signature l1)
  (T : Algebraic-Theory l2 σ)
  (G : UU l3)
  where

  data word-free-Algebra : UU (l1 ⊔ l3) where
    in-generator-word-free-Algebra : G → word-free-Algebra
    op-word-free-Algebra : is-model-of-signature-type σ word-free-Algebra

  data sim-word-free-Algebra : Relation (l1 ⊔ l2 ⊔ l3) word-free-Algebra where
    refl-sim-word-free-Algebra :
      is-reflexive sim-word-free-Algebra
    symmetric-sim-word-free-Algebra :
      is-symmetric sim-word-free-Algebra
    transitive-sim-word-free-Algebra :
      is-transitive sim-word-free-Algebra

    ap-op-sim-word-free-Algebra :
      (op : operation-signature σ)
      (x y : tuple word-free-Algebra (arity-operation-signature σ op)) →
      rel-tuple-Relation
        ( sim-word-free-Algebra)
        ( arity-operation-signature σ op)
        ( x)
        ( y) →
      sim-word-free-Algebra
        ( op-word-free-Algebra op x)
        ( op-word-free-Algebra op y)

    equation-sim-word-free-Algebra :
      (e : index-Algebraic-Theory σ T) →
      let
        (n , lhs , rhs) = index-abstract-equation-Algebraic-Theory σ T e
      in
        (v : fin-sequence word-free-Algebra n) →
        sim-word-free-Algebra
          ( eval-term σ op-word-free-Algebra v lhs)
          ( eval-term σ op-word-free-Algebra v rhs)

  trunc-prop-sim-word-free-Algebra :
    Relation-Prop (l1 ⊔ l2 ⊔ l3) word-free-Algebra
  trunc-prop-sim-word-free-Algebra =
    trunc-prop-Relation sim-word-free-Algebra

  rel-trunc-prop-sim-word-free-Algebra :
    Relation (l1 ⊔ l2 ⊔ l3) word-free-Algebra
  rel-trunc-prop-sim-word-free-Algebra =
    type-Relation-Prop trunc-prop-sim-word-free-Algebra

  equivalence-relation-word-free-Algebra :
    equivalence-relation (l1 ⊔ l2 ⊔ l3) word-free-Algebra
  equivalence-relation-word-free-Algebra =
    ( trunc-prop-sim-word-free-Algebra ,
      reflexive-trunc-prop-Relation
        ( sim-word-free-Algebra)
        ( refl-sim-word-free-Algebra) ,
      symmetric-trunc-prop-Relation
        ( sim-word-free-Algebra)
        ( symmetric-sim-word-free-Algebra) ,
      transitive-trunc-prop-Relation
        ( sim-word-free-Algebra)
        ( transitive-sim-word-free-Algebra))

  set-free-Algebra : Set (l1 ⊔ l2 ⊔ l3)
  set-free-Algebra =
    quotient-Set equivalence-relation-word-free-Algebra

  type-free-Algebra : UU (l1 ⊔ l2 ⊔ l3)
  type-free-Algebra = type-Set set-free-Algebra

  in-word-free-Algebra : word-free-Algebra → type-free-Algebra
  in-word-free-Algebra = quotient-map equivalence-relation-word-free-Algebra

  in-generator-free-Algebra : G → type-free-Algebra
  in-generator-free-Algebra g =
    in-word-free-Algebra (in-generator-word-free-Algebra g)

  abstract
    preserves-sim-op-word-free-Algebra :
      (op : operation-signature σ) →
      preserves-sim-equivalence-relation
        ( equivalence-relation-tuple
          ( equivalence-relation-word-free-Algebra)
          ( arity-operation-signature σ op))
        ( equivalence-relation-word-free-Algebra)
        ( op-word-free-Algebra op)
    preserves-sim-op-word-free-Algebra op {xs} {ys} xs|~|ys =
      map-trunc-Prop
        ( ap-op-sim-word-free-Algebra op xs ys)
        ( choice-rel-tuple-trunc-prop-Relation _ _ xs ys xs|~|ys)

  hom-op-word-free-Algebra :
    (op : operation-signature σ) →
    hom-equivalence-relation
      ( equivalence-relation-tuple
        ( equivalence-relation-word-free-Algebra)
        ( arity-operation-signature σ op))
      ( equivalence-relation-word-free-Algebra)
  hom-op-word-free-Algebra op =
    ( op-word-free-Algebra op ,
      preserves-sim-op-word-free-Algebra op)

  opaque
    op-free-Algebra : is-model-of-signature σ set-free-Algebra
    op-free-Algebra op =
      let
        n = arity-operation-signature σ op
      in
        map-is-set-quotient
          ( equivalence-relation-tuple equivalence-relation-word-free-Algebra n)
          ( tuple-Set set-free-Algebra n)
          ( reflecting-map-tuple-quotient-map
            ( equivalence-relation-word-free-Algebra)
            ( n))
          ( equivalence-relation-word-free-Algebra)
          ( set-free-Algebra)
          ( reflecting-map-quotient-map equivalence-relation-word-free-Algebra)
          ( is-set-quotient-tuple-set-quotient
            ( equivalence-relation-word-free-Algebra)
            ( n))
          ( is-set-quotient-set-quotient equivalence-relation-word-free-Algebra)
          ( hom-op-word-free-Algebra op)

  model-of-signature-free-Algebra : Model-Of-Signature (l1 ⊔ l2 ⊔ l3) σ
  model-of-signature-free-Algebra =
    ( set-free-Algebra , op-free-Algebra)

  abstract opaque
    unfolding op-free-Algebra

    compute-op-free-Algebra :
      (op : operation-signature σ)
      (v : tuple word-free-Algebra (arity-operation-signature σ op)) →
      op-free-Algebra op (map-tuple in-word-free-Algebra v) ＝
      in-word-free-Algebra (op-word-free-Algebra op v)
    compute-op-free-Algebra op =
      let
        n = arity-operation-signature σ op
      in
        coherence-square-map-is-set-quotient
          ( equivalence-relation-tuple equivalence-relation-word-free-Algebra n)
          ( tuple-Set set-free-Algebra n)
          ( reflecting-map-tuple-quotient-map
            ( equivalence-relation-word-free-Algebra)
            ( n))
          ( equivalence-relation-word-free-Algebra)
          ( set-free-Algebra)
          ( reflecting-map-quotient-map equivalence-relation-word-free-Algebra)
          ( is-set-quotient-tuple-set-quotient
            ( equivalence-relation-word-free-Algebra)
            ( n))
          ( is-set-quotient-set-quotient equivalence-relation-word-free-Algebra)
          ( hom-op-word-free-Algebra op)

  abstract
    compute-eval-term-free-Algebra :
      (k : ℕ) (t : term σ k) (v : fin-sequence word-free-Algebra k) →
      eval-term σ op-free-Algebra (in-word-free-Algebra ∘ v) t ＝
      in-word-free-Algebra (eval-term σ op-word-free-Algebra v t)

    Eq-compute-eval-tuple-term-free-Algebra :
      (k : ℕ) {n : ℕ} (t : tuple (term σ k) n)
      (v : fin-sequence word-free-Algebra k) →
      Eq-tuple
        ( n)
        ( eval-tuple-term σ op-free-Algebra (in-word-free-Algebra ∘ v) t)
        ( map-tuple
          ( in-word-free-Algebra)
          ( eval-tuple-term σ op-word-free-Algebra v t))

    compute-eval-tuple-term-free-Algebra :
      (k : ℕ) {n : ℕ} (t : tuple (term σ k) n)
      (v : fin-sequence word-free-Algebra k) →
      eval-tuple-term σ op-free-Algebra (in-word-free-Algebra ∘ v) t ＝
      map-tuple
        ( in-word-free-Algebra)
        ( eval-tuple-term σ op-word-free-Algebra v t)

    compute-eval-term-free-Algebra k (var-term i) v = refl
    compute-eval-term-free-Algebra k (op-term op xs) v =
      ( ap
        ( op-free-Algebra op)
        ( compute-eval-tuple-term-free-Algebra k xs v)) ∙
      ( compute-op-free-Algebra
        ( op)
        ( eval-tuple-term σ op-word-free-Algebra v xs))

    Eq-compute-eval-tuple-term-free-Algebra k {0} empty-tuple v = map-raise star
    Eq-compute-eval-tuple-term-free-Algebra k {succ-ℕ n} (x ∷ xs) v =
      ( compute-eval-term-free-Algebra k x v ,
        Eq-compute-eval-tuple-term-free-Algebra k xs v)

    compute-eval-tuple-term-free-Algebra k xs v =
      eq-Eq-tuple _ _ _ (Eq-compute-eval-tuple-term-free-Algebra k xs v)

    is-algebra-free-Algebra :
      is-algebra-Model-of-Signature σ T model-of-signature-free-Algebra
    is-algebra-free-Algebra i v =
      let
        (k , lhs , rhs) = index-abstract-equation-Algebraic-Theory σ T i
      in
        ind-is-set-quotient
          ( equivalence-relation-fin-sequence-equivalence-relation
            ( equivalence-relation-word-free-Algebra)
            ( k))
          ( fin-sequence-Set set-free-Algebra k)
          ( reflecting-quotient-map-fin-sequence
            ( equivalence-relation-word-free-Algebra)
            ( k))
          ( is-set-quotient-fin-sequence-set-quotient
            ( equivalence-relation-word-free-Algebra)
            ( k))
          ( λ v →
            Id-Prop
              ( set-free-Algebra)
              ( eval-term σ op-free-Algebra v lhs)
              ( eval-term σ op-free-Algebra v rhs))
          ( λ a →
            equational-reasoning
              eval-term σ op-free-Algebra (in-word-free-Algebra ∘ a) lhs
              ＝ in-word-free-Algebra (eval-term σ op-word-free-Algebra a lhs)
                by compute-eval-term-free-Algebra k lhs a
              ＝ in-word-free-Algebra (eval-term σ op-word-free-Algebra a rhs)
                by
                  apply-effectiveness-quotient-map'
                    ( equivalence-relation-word-free-Algebra)
                    ( unit-trunc-Prop (equation-sim-word-free-Algebra i a))
              ＝ eval-term σ op-free-Algebra (in-word-free-Algebra ∘ a) rhs
                by inv (compute-eval-term-free-Algebra k rhs a))
          ( v)

  free-Algebra : Algebra (l1 ⊔ l2 ⊔ l3) σ T
  free-Algebra =
    ( model-of-signature-free-Algebra ,
      is-algebra-free-Algebra)
```

## Properties

### The universal property of the free algebra

```agda
module _
  {l1 l2 l3 : Level}
  (σ : signature l1)
  (T : Algebraic-Theory l2 σ)
  (G : UU l3)
  where

  is-free-Algebra :
    {l4 : Level} (l5 : Level) → Algebra l4 σ T →
    UU (l1 ⊔ l2 ⊔ l3 ⊔ l4 ⊔ lsuc l5)
  is-free-Algebra l5 A =
    (B : Algebra l5 σ T) → (G → type-Algebra σ T B) ≃ hom-Algebra σ T A B

module _
  {l1 l2 l3 : Level}
  (σ : signature l1)
  (T : Algebraic-Theory l2 σ)
  (G : UU l3)
  {l4 : Level}
  (B : Algebra l4 σ T)
  where

  map-word-is-free-free-Algebra :
    (G → type-Algebra σ T B) → word-free-Algebra σ T G → type-Algebra σ T B

  -- necessarily preferred to map-tuple for termination checking
  map-tuple-word-is-free-free-Algebra :
    (G → type-Algebra σ T B) →
    {n : ℕ} →
    tuple (word-free-Algebra σ T G) n →
    tuple (type-Algebra σ T B) n

  map-word-is-free-free-Algebra f (in-generator-word-free-Algebra x) = f x
  map-word-is-free-free-Algebra f (op-word-free-Algebra op xs) =
    is-model-set-Algebra σ T B
      ( op)
      ( map-tuple-word-is-free-free-Algebra f xs)

  map-tuple-word-is-free-free-Algebra f empty-tuple = empty-tuple
  map-tuple-word-is-free-free-Algebra f (w ∷ ws) =
    map-word-is-free-free-Algebra f w ∷
    map-tuple-word-is-free-free-Algebra f ws

  abstract
    Eq-map-tuple-map-tuple-word-is-free-free-Algebra :
      (f : G → type-Algebra σ T B)
      {n : ℕ} (w : tuple (word-free-Algebra σ T G) n) →
      Eq-tuple
        ( n)
        ( map-tuple-word-is-free-free-Algebra f w)
        ( map-tuple (map-word-is-free-free-Algebra f) w)
    Eq-map-tuple-map-tuple-word-is-free-free-Algebra f empty-tuple =
      map-raise star
    Eq-map-tuple-map-tuple-word-is-free-free-Algebra f (w ∷ ws) =
      ( refl ,
        Eq-map-tuple-map-tuple-word-is-free-free-Algebra f ws)

    eq-map-tuple-map-tuple-word-is-free-free-Algebra :
      (f : G → type-Algebra σ T B)
      {n : ℕ} (w : tuple (word-free-Algebra σ T G) n) →
      map-tuple-word-is-free-free-Algebra f w ＝
      map-tuple (map-word-is-free-free-Algebra f) w
    eq-map-tuple-map-tuple-word-is-free-free-Algebra f w =
      eq-Eq-tuple _ _ _ (Eq-map-tuple-map-tuple-word-is-free-free-Algebra f w)

  abstract
    eval-term-map-word-is-free-free-Algebra :
      (f : G → type-Algebra σ T B) {n : ℕ} (t : term σ n)
      (v : fin-sequence (word-free-Algebra σ T G) n) →
      map-word-is-free-free-Algebra f (eval-term σ op-word-free-Algebra v t) ＝
      eval-term
        ( σ)
        ( is-model-set-Algebra σ T B)
        ( map-word-is-free-free-Algebra f ∘ v) t

    eval-tuple-term-map-tuple-word-is-free-free-Algebra :
      (f : G → type-Algebra σ T B) {k n : ℕ} (t : tuple (term σ k) n)
      (v : fin-sequence (word-free-Algebra σ T G) k) →
      Eq-tuple
        ( n)
        ( map-tuple-word-is-free-free-Algebra
          ( f)
          ( eval-tuple-term σ op-word-free-Algebra v t))
        ( eval-tuple-term
          ( σ)
          ( is-model-set-Algebra σ T B)
          ( map-word-is-free-free-Algebra f ∘ v)
          ( t))

    eval-term-map-word-is-free-free-Algebra f (var-term x) v =
      refl
    eval-term-map-word-is-free-free-Algebra f (op-term op x) v =
      ap
        ( is-model-set-Algebra σ T B op)
        ( eq-Eq-tuple _ _ _
          ( eval-tuple-term-map-tuple-word-is-free-free-Algebra f x v))

    eval-tuple-term-map-tuple-word-is-free-free-Algebra f empty-tuple v =
      map-raise star
    eval-tuple-term-map-tuple-word-is-free-free-Algebra f (x ∷ xs) v =
      ( eval-term-map-word-is-free-free-Algebra f x v ,
        eval-tuple-term-map-tuple-word-is-free-free-Algebra f xs v)

  abstract
    preserves-sim-map-word-is-free-free-Algebra' :
      (f : G → type-Algebra σ T B) (x y : word-free-Algebra σ T G) →
      sim-word-free-Algebra σ T G x y →
      map-word-is-free-free-Algebra f x ＝ map-word-is-free-free-Algebra f y

    preserves-sim-map-tuple-word-is-free-free-Algebra' :
      (f : G → type-Algebra σ T B) {n : ℕ}
      (x y : tuple (word-free-Algebra σ T G) n) →
      rel-tuple-Relation (sim-word-free-Algebra σ T G) n x y →
      Eq-tuple
        ( n)
        ( map-tuple-word-is-free-free-Algebra f x)
        ( map-tuple-word-is-free-free-Algebra f y)

    preserves-sim-map-word-is-free-free-Algebra'
      f x .x (refl-sim-word-free-Algebra .x) =
      refl
    preserves-sim-map-word-is-free-free-Algebra'
      f x y (symmetric-sim-word-free-Algebra .y .x y~x) =
      inv (preserves-sim-map-word-is-free-free-Algebra' f y x y~x)
    preserves-sim-map-word-is-free-free-Algebra'
      f x y (transitive-sim-word-free-Algebra .x z .y z~y x~z) =
      ( preserves-sim-map-word-is-free-free-Algebra' f x z x~z) ∙
      ( preserves-sim-map-word-is-free-free-Algebra' f z y z~y)
    preserves-sim-map-word-is-free-free-Algebra'
      f (op-word-free-Algebra op x) (op-word-free-Algebra .op y)
      (ap-op-sim-word-free-Algebra .op .x .y x~y) =
      ap
        ( is-model-set-Algebra σ T B op)
        ( eq-Eq-tuple _ _ _
          ( preserves-sim-map-tuple-word-is-free-free-Algebra' f x y x~y))
    preserves-sim-map-word-is-free-free-Algebra'
      f x y (equation-sim-word-free-Algebra e v) =
      let
        (k , lhs , rhs) = index-abstract-equation-Algebraic-Theory σ T e
      in
        ( eval-term-map-word-is-free-free-Algebra f lhs v) ∙
        ( is-algebra-model-Algebra σ T B
          ( e)
          ( map-word-is-free-free-Algebra f ∘ v)) ∙
        ( inv (eval-term-map-word-is-free-free-Algebra f rhs v))

    preserves-sim-map-tuple-word-is-free-free-Algebra'
      f empty-tuple empty-tuple _ =
      map-raise star
    preserves-sim-map-tuple-word-is-free-free-Algebra'
      f (x ∷ xs) (y ∷ ys) (x~y , xs~ys) =
      ( preserves-sim-map-word-is-free-free-Algebra' f x y x~y ,
        preserves-sim-map-tuple-word-is-free-free-Algebra' f xs ys xs~ys)

  preserves-sim-map-word-is-free-free-Algebra :
    (f : G → type-Algebra σ T B) →
    preserves-sim-equivalence-relation
      ( equivalence-relation-word-free-Algebra σ T G)
      ( Id-equivalence-relation (set-Algebra σ T B))
      ( map-word-is-free-free-Algebra f)
  preserves-sim-map-word-is-free-free-Algebra f {x} {y} =
    rec-trunc-Prop
      ( Id-Prop
        ( set-Algebra σ T B)
        ( map-word-is-free-free-Algebra f x)
        ( map-word-is-free-free-Algebra f y))
      ( preserves-sim-map-word-is-free-free-Algebra' f x y)
```
