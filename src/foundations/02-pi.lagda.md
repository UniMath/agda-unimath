-- Section 2.3 The identity function, composition, and their laws

-- Definition 2.3.1
id : {i : Level} {A : UU i} → A → A
id a = a 

-- Dependent version of Definition 2.3.2
_∘_ : {i j k : Level}
     → {A : UU i} {B : A → UU j} {C : (a : A) → B a → UU k}
     → ({a : A} → (b : B a) → C a b) → (f : (a : A) → B a) → (a : A) → C a (f a)
(g ∘ f) a = g (f a)

-- Exercise 2.3
const : {i j : Level} (A : UU i) (B : UU j) (b : B) → A → B

-- Exercise 2.4
Π-swap : {i j k : Level} {A : UU i} {B : UU j} {C : A → (B → UU k)} →
  ((x : A) (y : B) → C x y) → ((y : B) (x : A) → C x y)
Π-swap f y x = f x y
